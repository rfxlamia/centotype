1//! Session management with thread-safe state tracking and performance monitoring
2use crate::types::*;
3use chrono::{DateTime, Utc};
4use parking_lot::{Mutex, RwLock};
5use std::collections::HashMap;
6use std::sync::Arc;
7use std::time::{Duration, Instant};
8use tracing::{debug, error, info, warn};

10/// Thread-safe session manager with performance tracking
11pub struct SessionManager {
12    current_session: Arc<RwLock<Option<SessionState>>>,
13    sessions: Arc<Mutex<HashMap<uuid::Uuid, SessionState>>>,
14    performance_tracker: Arc<Mutex<SessionPerformanceTracker>>,
15}
16
17impl SessionManager {
18    pub fn new() -> Self {
19        Self {
20            current_session: Arc::new(RwLock::new(None)),
21            sessions: Arc::new(Mutex::new(HashMap::new())),
22            performance_tracker: Arc::new(Mutex::new(SessionPerformanceTracker::new())),
23        }
24    }
25
26    /// Start a new typing session with comprehensive state initialization
27    pub fn start_session(&mut self, mut session_state: SessionState) -> Result<()> {
28        let start_time = Instant::now();
29
30        // Validate session state
31        if session_state.target_text.is_empty() {
32            return Err(CentotypeError::State("Target text cannot be empty".to_string()));
33        }
34
35        // Ensure consistent initial state
36        session_state.typed_text.clear();
37        session_state.cursor_position = 0;
38        session_state.keystrokes.clear();
39        session_state.is_paused = false;
40        session_state.is_completed = false;
41        session_state.started_at = Utc::now();
42        session_state.paused_duration = Duration::default();
43
44        let session_id = session_state.session_id;
45
46        // Store session and set as current
47        {
48            let mut sessions = self.sessions.lock();
49            sessions.insert(session_id, session_state.clone());
50        }
51
52        {
53            let mut current = self.current_session.write();
54            *current = Some(session_state);
55        }
56
57        // Track performance
58        let elapsed = start_time.elapsed();
59        self.performance_tracker.lock().record_session_start(elapsed);
60
61        info!(session_id = %session_id, "Started new typing session");
62        Ok(())
63    }
64
65    /// Get current session state (read-only)
66    pub fn current_state(&self) -> Result<SessionState> {
67        let current = self.current_session.read();
68        current
69            .as_ref()
70            .map(|s| s.clone())
71            .ok_or_else(|| CentotypeError::State("No active session".to_string()))
72    }
73
74    /// Check if there's an active session
75    pub fn has_active_session(&self) -> bool {
76        self.current_session.read().is_some()
77    }
78
79    /// Get session by ID
80    pub fn get_session(&self, session_id: uuid::Uuid) -> Result<SessionState> {
81        let sessions = self.sessions.lock();
82        sessions
83            .get(&session_id)
84            .cloned()
85            .ok_or_else(|| CentotypeError::State(format!("Session {} not found", session_id)))
86    }
87
88    /// Update session state with atomic operations
89    pub fn update_state(&mut self, update: StateUpdate) -> Result<()> {
90        let start_time = Instant::now();
91
92        let mut current = self.current_session.write();
93        let session = current
94            .as_mut()
95            .ok_or_else(|| CentotypeError::State("No active session".to_string()))?;
96
97        // Apply state update
98        match update {
99            StateUpdate::AddKeystroke(keystroke) => {
100                self.apply_keystroke(session, keystroke)?;
101            }
102            StateUpdate::SetPaused(paused) => {
103                self.set_paused_state(session, paused)?;
104            }
105            StateUpdate::MoveCursor(position) => {
106                self.move_cursor(session, position)?;
107            }
108            StateUpdate::Complete => {
109                self.complete_session_internal(session)?;
110            }
111        }
112
113        // Update stored session
114        let session_id = session.session_id;
115        {
116            let mut sessions = self.sessions.lock();
117            sessions.insert(session_id, session.clone());
118        }
119
120        // Track performance
121        let elapsed = start_time.elapsed();
122        self.performance_tracker.lock().record_state_update(elapsed);
123
124        debug!(session_id = %session_id, "Updated session state");
125        Ok(())
126    }
127
128    /// Reset session manager state
129    pub fn reset(&mut self) {
130        let mut current = self.current_session.write();
131        *current = None;
132
133        // Clear performance tracking
134        *self.performance_tracker.lock() = SessionPerformanceTracker::new();
135
136        info!("Reset session manager");
137    }
138
139    /// Complete the current session
140    pub fn complete_current_session(&mut self) -> Result<SessionState> {
141        let mut current = self.current_session.write();
142        let session = current
143            .as_mut()
144            .ok_or_else(|| CentotypeError::State("No active session to complete".to_string()))?;
145
146        self.complete_session_internal(session)?;
147        let completed_session = session.clone();
148
149        // Update stored session
150        let session_id = session.session_id;
151        {
152            let mut sessions = self.sessions.lock();
153            sessions.insert(session_id, session.clone());
154        }
155
156        info!(session_id = %session_id, "Completed typing session");
157        Ok(completed_session)
158    }
159
160    /// Get performance metrics for monitoring
161    pub fn get_performance_metrics(&self) -> SessionManagerMetrics {
162        self.performance_tracker.lock().get_metrics()
163    }

165    // Private helper methods
166
167    fn apply_keystroke(&self, session: &mut SessionState, keystroke: Keystroke) -> Result<()> {
168        // Validate keystroke timing
169        if keystroke.timestamp < session.started_at {
170            return Err(CentotypeError::State("Invalid keystroke timestamp".to_string()));
171        }
172
173        // Handle different keystroke types
174        match keystroke.char_typed {
175            Some(ch) => {
176                if keystroke.is_correction {
177                    self.handle_correction(session, ch)?;
178                } else {
179                    self.handle_regular_input(session, ch)?;
180                }
181            }
182            None => {
183                // Handle special keys (backspace, etc.)
184                self.handle_special_key(session)?;
185            }
186        }
187
188        // Record keystroke
189        session.keystrokes.push(keystroke);
190        Ok(())
191    }
192
193    fn handle_regular_input(&self, session: &mut SessionState, ch: char) -> Result<()> {
194        // Validate cursor position
195        if session.cursor_position > session.target_text.len() {
196            return Err(CentotypeError::State("Cursor position out of bounds".to_string()));
197        }
198
199        // Add character to typed text
200        if session.cursor_position == session.typed_text.len() {
201            session.typed_text.push(ch);
202        } else {
203            // Insert character at cursor position
204            let mut chars: Vec<char> = session.typed_text.chars().collect();
205            if session.cursor_position <= chars.len() {
206                chars.insert(session.cursor_position, ch);
207                session.typed_text = chars.into_iter().collect();
208            }
209        }
210
211        session.cursor_position += 1;
212        Ok(())
213    }
214
215    fn handle_correction(&self, session: &mut SessionState, _ch: char) -> Result<()> {
216        // Handle correction logic - this would be more complex in practice
217        if session.cursor_position > 0 {
218            session.cursor_position -= 1;
219            if session.cursor_position < session.typed_text.len() {
220                let mut chars: Vec<char> = session.typed_text.chars().collect();
221                chars.remove(session.cursor_position);
222                session.typed_text = chars.into_iter().collect();
223            }
224        }
225        Ok(())
226    }
227
228    fn handle_special_key(&self, session: &mut SessionState) -> Result<()> {
229        // Handle backspace and other special keys
230        if session.cursor_position > 0 && !session.typed_text.is_empty() {
231            session.cursor_position -= 1;
232            if session.cursor_position < session.typed_text.len() {
233                let mut chars: Vec<char> = session.typed_text.chars().collect();
234                chars.remove(session.cursor_position);
235                session.typed_text = chars.into_iter().collect();
236            }
237        }
238        Ok(())
239    }
240
241    fn set_paused_state(&self, session: &mut SessionState, paused: bool) -> Result<()> {
242        if session.is_completed {
243            return Err(CentotypeError::State("Cannot pause completed session".to_string()));
244        }
245
246        if session.is_paused == paused {
247            return Ok(()); // No change needed
248        }
249
250        if paused {
251            session.is_paused = true;
252            debug!("Session paused");
253        } else {
254            session.is_paused = false;
255            debug!("Session resumed");
256        }
257
258        Ok(())
259    }
260
261    fn move_cursor(&self, session: &mut SessionState, position: usize) -> Result<()> {
262        if position > session.target_text.len() {
263            return Err(CentotypeError::State("Cursor position out of bounds".to_string()));
264        }
265
266        session.cursor_position = position;
267        Ok(())
268    }
269
270    fn complete_session_internal(&self, session: &mut SessionState) -> Result<()> {
271        if session.is_completed {
272            return Ok(()); // Already completed
273        }
274
275        session.is_completed = true;
276        session.is_paused = false;
277
278        info!(session_id = %session.session_id, "Session marked as completed");
279        Ok(())
280    }
281}

305/// Performance tracking for session operations
306#[derive(Debug, Clone)]
307struct SessionPerformanceTracker {
308    session_starts: Vec<Duration>,
309    state_updates: Vec<Duration>,
310    total_operations: u64,
311}
312
313impl SessionPerformanceTracker {
314    fn new() -> Self {
315        Self {
316            session_starts: Vec::new(),
317            state_updates: Vec::new(),
318            total_operations: 0,
319        }
320    }
321
322    fn record_session_start(&mut self, duration: Duration) {
323        self.session_starts.push(duration);
324        self.total_operations += 1;
325
326        // Keep only recent measurements for memory efficiency
327        if self.session_starts.len() > 1000 {
328            self.session_starts.remove(0);
329        }
330    }
331
332    fn record_state_update(&mut self, duration: Duration) {
333        self.state_updates.push(duration);
334        self.total_operations += 1;
335
336        // Keep only recent measurements
337        if self.state_updates.len() > 10000 {
338            self.state_updates.remove(0);
339        }
340    }
341
342    fn get_metrics(&self) -> SessionManagerMetrics {
343        SessionManagerMetrics {
344            avg_session_start_time: self.calculate_average(&self.session_starts),
345            avg_state_update_time: self.calculate_average(&self.state_updates),
346            p95_state_update_time: self.calculate_percentile(&self.state_updates, 0.95),
347            p99_state_update_time: self.calculate_percentile(&self.state_updates, 0.99),
348            total_operations: self.total_operations,
349        }
350    }
351
352    fn calculate_average(&self, durations: &[Duration]) -> Duration {
353        if durations.is_empty() {
354            return Duration::default();
355        }
356
357        let sum: Duration = durations.iter().sum();
358        sum / durations.len() as u32
359    }
360
361    fn calculate_percentile(&self, durations: &[Duration], percentile: f64) -> Duration {
362        if durations.is_empty() {
363            return Duration::default();
364        }
365
366        let mut sorted: Vec<Duration> = durations.to_vec();
367        sorted.sort();
368
369        let index = ((durations.len() as f64 - 1.0) * percentile) as usize;
370        sorted[index.min(sorted.len() - 1)]
371    }
372}

374/// Metrics for session manager performance monitoring
375#[derive(Debug, Clone)]
376pub struct SessionManagerMetrics {
377    pub avg_session_start_time: Duration,
378    pub avg_state_update_time: Duration,
379    pub p95_state_update_time: Duration,
380    pub p99_state_update_time: Duration,
381    pub total_operations: u64,
382}

388    }
