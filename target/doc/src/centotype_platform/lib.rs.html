1//! # Centotype Platform
2//!
3//! Platform-specific integrations and optimizations for different operating systems
4//! and terminal environments. This crate provides:
5//!
6//! - Terminal capability detection and optimization
7//! - OS-specific performance optimizations
8//! - Cross-platform terminal handling abstractions
9//! - Hardware detection and feature availability
10//!
11//! ## Supported Platforms
12//!
13//! - **Linux**: Full support with optimizations for common terminals
14//! - **macOS**: Native terminal integration with performance tuning
15//! - **Windows**: Windows Terminal, PowerShell, and cmd.exe support
16//!
17//! ## Performance Optimizations
18//!
19//! - Terminal-specific render optimizations
20//! - Platform-native input handling
21//! - Memory management tuning per OS
22//! - CPU architecture detection

39/// Main platform manager that coordinates all platform-specific functionality
40pub struct PlatformManager {
41    info: PlatformInfo,
42    terminal_caps: TerminalCapabilities,
43    input_optimizations: InputOptimizations,
44    performance_settings: PlatformPerformance,
45}
46
47impl PlatformManager {
48    pub fn new() -> Result<Self> {
49        let detector = PlatformDetector::new();
50        let info = detector.detect_platform()?;
51        let terminal_caps = detector.detect_terminal_capabilities()?;
52        let input_optimizations = InputOptimizations::for_platform(&info)?;
53        let performance_settings = PlatformPerformance::optimize_for_platform(&info)?;
54
55        tracing::info!(
56            "Platform detected: OS={:?}, Terminal={:?}, Arch={:?}",
57            info.os_type,
58            terminal_caps.terminal_type,
59            info.architecture
60        );
61
62        Ok(Self {
63            info,
64            terminal_caps,
65            input_optimizations,
66            performance_settings,
67        })
68    }
69
70    pub fn platform_info(&self) -> &PlatformInfo {
71        &self.info
72    }
73
74    pub fn terminal_capabilities(&self) -> &TerminalCapabilities {
75        &self.terminal_caps
76    }
77
78    pub fn input_optimizations(&self) -> &InputOptimizations {
79        &self.input_optimizations
80    }
81
82    pub fn performance_settings(&self) -> &PlatformPerformance {
83        &self.performance_settings
84    }
85
86    /// Check if the platform meets minimum requirements
87    pub fn validate_platform(&self) -> PlatformValidation {
88        let mut issues = Vec::new();
89        let mut warnings = Vec::new();
90
91        // Check terminal capabilities
92        if !self.terminal_caps.supports_raw_mode {
93            issues.push("Terminal does not support raw mode".to_string());
94        }
95
96        if !self.terminal_caps.supports_color {
97            warnings.push("Terminal does not support color - using monochrome mode".to_string());
98        }
99
100        // Check performance capabilities
101        if !self.performance_settings.can_meet_targets {
102            warnings.push("Platform may not meet performance targets".to_string());
103        }
104
105        // Check input handling
106        if !self.input_optimizations.high_precision_timing {
107            warnings.push("High precision input timing not available".to_string());
108        }
109
110        PlatformValidation {
111            is_supported: issues.is_empty(),
112            issues,
113            warnings,
114            recommended_settings: self.get_recommended_settings(),
115        }
116    }
117
118    fn get_recommended_settings(&self) -> RecommendedSettings {
119        RecommendedSettings {
120            use_reduced_effects: !self.performance_settings.can_meet_targets,
121            buffer_size: self.input_optimizations.recommended_buffer_size,
122            render_frequency: self.performance_settings.recommended_fps,
123            memory_limit_mb: self.performance_settings.memory_limit_mb,
124        }
125    }
126
127    /// Apply platform-specific optimizations
128    pub fn apply_optimizations(&self) -> Result<()> {
129        // Apply input optimizations
130        self.input_optimizations.configure_input_system()?;
131
132        // Apply performance optimizations
133        self.performance_settings.configure_performance()?;
134
135        // Configure terminal-specific settings
136        self.terminal_caps.configure_terminal()?;
137
138        tracing::info!("Platform optimizations applied successfully");
139        Ok(())
140    }
141
142    /// Get system resource information
143    pub fn get_system_metrics(&self) -> SystemMetrics {
144        self.performance_settings.get_current_metrics()
145    }
146
147    /// Check if graceful degradation is needed
148    pub fn should_use_fallback_mode(&self) -> bool {
149        !self.performance_settings.can_meet_targets
150            || self.terminal_caps.has_limitations()
151    }
152
153    /// Get platform-specific error recovery strategies
154    pub fn get_error_recovery(&self) -> ErrorRecoveryStrategy {
155        match self.info.os_type {
156            OSType::Windows => ErrorRecoveryStrategy {
157                restore_console_mode: true,
158                flush_input_buffer: true,
159                reset_cursor_visibility: true,
160                clear_screen_on_exit: false,
161            },
162            OSType::MacOS | OSType::Linux => ErrorRecoveryStrategy {
163                restore_console_mode: true,
164                flush_input_buffer: false,
165                reset_cursor_visibility: true,
166                clear_screen_on_exit: true,
167            },
168            OSType::Unknown => ErrorRecoveryStrategy::default(),
169        }
170    }
171}

173#[derive(Debug, Clone)]
174pub struct PlatformValidation {
175    pub is_supported: bool,
176    pub issues: Vec<String>,
177    pub warnings: Vec<String>,
178    pub recommended_settings: RecommendedSettings,
179}
180
181#[derive(Debug, Clone)]
182pub struct RecommendedSettings {
183    pub use_reduced_effects: bool,
184    pub buffer_size: usize,
185    pub render_frequency: u32,
186    pub memory_limit_mb: u64,
187}
188
189#[derive(Debug, Clone)]
190pub struct ErrorRecoveryStrategy {
191    pub restore_console_mode: bool,
192    pub flush_input_buffer: bool,
193    pub reset_cursor_visibility: bool,
194    pub clear_screen_on_exit: bool,
195}
196
197impl Default for ErrorRecoveryStrategy {
198    fn default() -> Self {
199        Self {
200            restore_console_mode: true,
201            flush_input_buffer: true,
202            reset_cursor_visibility: true,
203            clear_screen_on_exit: true,
204        }
205    }
206}
